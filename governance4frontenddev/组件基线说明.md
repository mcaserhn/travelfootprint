# 1) `components/ui/Button.tsx`（强约束：只用品牌 tokens）

```tsx
import * as React from "react";

type ButtonVariant = "primary" | "dark" | "ghost" | "outline";
type ButtonSize = "sm" | "md" | "lg";

export type ButtonProps = React.ButtonHTMLAttributes<HTMLButtonElement> & {
  variant?: ButtonVariant;
  size?: ButtonSize;
  fullWidth?: boolean;
};

const base =
  "inline-flex items-center justify-center gap-2 rounded-xl transition " +
  "focus:outline-none focus:ring-2 focus:ring-aden-digital-blue focus:ring-offset-2 " +
  "disabled:opacity-50 disabled:pointer-events-none";

const sizes: Record<ButtonSize, string> = {
  sm: "h-9 px-4 text-sm",
  md: "h-11 px-6 text-base",
  lg: "h-12 px-7 text-base",
};

const variants: Record<ButtonVariant, string> = {
  primary: "bg-aden-orange text-aden-white hover:opacity-90",
  dark: "bg-aden-dark-blue text-aden-white hover:opacity-90",
  ghost: "bg-transparent text-aden-dark-blue hover:bg-aden-light-grey",
  outline:
    "bg-transparent text-aden-dark-blue border border-aden-mid-grey hover:bg-aden-light-grey",
};

export function Button({
  className = "",
  variant = "primary",
  size = "md",
  fullWidth = false,
  type = "button",
  ...props
}: ButtonProps) {
  return (
    <button
      type={type}
      className={[
        base,
        sizes[size],
        variants[variant],
        fullWidth ? "w-full" : "",
        className,
      ].join(" ")}
      {...props}
    />
  );
}
```

✅ 这个组件强制把“按钮”收敛成 4 种语义：

*   `primary`（品牌橙）
*   `dark`（深蓝）
*   `ghost`（透明底）
*   `outline`（边框）

# 2) `components/ui/Card.tsx`（官网模块通用容器）

```tsx
import * as React from "react";

export type CardProps = React.HTMLAttributes<HTMLDivElement> & {
  variant?: "default" | "soft" | "dark";
};

const variants: Record<NonNullable<CardProps["variant"]>, string> = {
  default: "bg-aden-white border border-aden-mid-grey",
  soft: "bg-aden-light-grey border border-aden-mid-grey",
  dark: "bg-aden-dark-blue border border-aden-dark-blue text-aden-white",
};

export function Card({
  className = "",
  variant = "default",
  ...props
}: CardProps) {
  return (
    <div
      className={[
        "rounded-2xl p-6 md:p-8",
        variants[variant],
        className,
      ].join(" ")}
      {...props}
    />
  );
}

export function CardTitle({
  className = "",
  ...props
}: React.HTMLAttributes<HTMLHeadingElement>) {
  return (
    <h3
      className={["font-title text-xl text-aden-dark-blue", className].join(" ")}
      {...props}
    />
  );
}

export function CardText({
  className = "",
  ...props
}: React.HTMLAttributes<HTMLParagraphElement>) {
  return (
    <p
      className={["mt-2 text-aden-dark-grey", className].join(" ")}
      {...props}
    />
  );
}
```

# 3) `components/ui/Section.tsx`（强制统一容器宽度）

```tsx
import * as React from "react";

export type SectionProps = React.HTMLAttributes<HTMLElement> & {
  tone?: "default" | "light" | "dark";
  size?: "sm" | "md" | "lg";
};

const toneMap = {
  default: "bg-aden-white text-aden-dark-grey",
  light: "bg-aden-light-grey text-aden-dark-grey",
  dark: "bg-aden-dark-blue text-aden-white",
} as const;

const sizeMap = {
  sm: "py-12 md:py-16",
  md: "py-16 md:py-20",
  lg: "py-20 md:py-28",
} as const;

export function Section({
  className = "",
  tone = "default",
  size = "md",
  ...props
}: SectionProps) {
  return (
    <section
      className={[toneMap[tone], sizeMap[size], className].join(" ")}
      {...props}
    />
  );
}

export function Container({
  className = "",
  ...props
}: React.HTMLAttributes<HTMLDivElement>) {
  return (
    <div
      className={["mx-auto max-w-content px-4", className].join(" ")}
      {...props}
    />
  );
}

export function SectionHeader({
  title,
  description,
}: {
  title: string;
  description?: string;
}) {
  return (
    <div className="max-w-2xl">
      <h2 className="font-title text-3xl md:text-4xl">{title}</h2>
      {description ? (
        <p className="mt-4 max-w-prose opacity-90">{description}</p>
      ) : null}
    </div>
  );
}
```

> `Container` 统一宽度 `max-w-content`，配合你之前的 Tailwind tokens（`maxWidth.content = 1200px`）就能全站一致。

# 4) `components/motion/FadeIn.tsx`（只在 Client 用，避免 Server Component 误用）

```tsx
"use client";

import * as React from "react";
import { motion } from "framer-motion";

export function FadeIn({ children }: { children: React.ReactNode }) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 24 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{ once: true, margin: "-80px" }}
      transition={{ duration: 0.6, ease: "easeOut" }}
    >
      {children}
    </motion.div>
  );
}
```

# 5) `styles/globals.css`（把“不要 justify”固化）

```css
@tailwind base;
@tailwind components;
@tailwind utilities;

@layer base {
  /* 全站默认字体（也可在 layout.tsx 用 className="font-body"） */
  html {
    font-family: Catamaran, Arial, sans-serif;
  }

  /* 品牌排版：禁止 justify，统一左对齐 */
  p {
    text-align: left;
  }
}
```

# 7) 你现在可以怎么“应用”这套模板（最短路径）

1.  把以上 4 个组件文件复制到 `components/`
2.  确保 `tailwind.config.js` 已存在 `aden.*` colors、`fontFamily.title/body/cn`、`maxWidth.content`
3.  在 `app/[locale]/layout.tsx` 里把 body 固定为：

```tsx
<body className="font-body bg-aden-white text-aden-dark-grey">{children}</body>
```

然后你们页面开发就变成“拼积木”：

> **Section + Container + SectionHeader + Card + Button + FadeIn**
